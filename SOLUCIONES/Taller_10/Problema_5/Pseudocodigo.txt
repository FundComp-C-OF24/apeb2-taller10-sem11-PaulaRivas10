4.			<< matriz[i][j], " | "

7.	FinPara

7.	FinPara
8.FinFuncion

1.Funcion restarMatriz(matriz , filas , columnas)
2.	Definir resta Como Entero
3.	Para(i=0; i<filas; i=i+1)
4.		Para(j=0; j<columnas; j=j+1)
5.			Si(i=0 y j=0)
6.				resta= matriz[i][j]
7.			Sino
8.				resta = resta - matriz[i][j]
9.			FinSi		
10		FinPara
11.	FinPara
12. Devolver resta
13.FinFuncion
